Comparison of Scala with Java

The Basics

Package

Java


package myPackage

Scala


package myPackage

Folder structure

  • Java:

    • Folder structure matches package structure

  • Scala:

    • Any folder structure. Unrelated to package structure

    Any file can be in any package regardless of folder

Imports

Java


import java.util.Map
import java.util.List
import java.io.*

Scala


import java.util.map
import java.util.List
import java.io._
import java.lang.reflect.{Member,Field,Method}

Immutable variable declaration

Java


final String name = "Shusaku"

Scala


val name : String = "Shusaku"

Mutable variable declaration

Java


String name = "Shusaku"
name = "Go Seigen"

Scala


var name : String = "Shusaku"
name = "Go Seigen"

Java Bean



package myPackage

import myPackage.mySubPackage
import java.util.Objects

class Person {

  private String givenName  = null ;
  private String familyName = null ;

  public Person( givenName : String , familyName : String ) {
    this.givenName = givenName ;
    this.familyName = familyName ;
  }

  public String getGivenName() {
    return this.givenName ;
  }

  public String getFamilyName() {
    return this.familyName ;
  }

  public String toString() {
    return this.givenName + " " + this.familyName ;
  }

  @Override
  public int hashCode() {
    return Objects.hashCode(this.firstName, this.lastName);
  }

  public boolean equals( Object aThat ) {
    boolean isEqual = true ;
    if( aThat == null ) {
      isEquals = false ;
    } else {
      if ( this != aThat ) {
        if ( !(aThat instanceof Person) ) {
          isEquals = false ;
        } else {
          Person that = (Person)aThat;
          if( that.hashCode() != hashCode() ) {
            isEquals = false ;
          }
        }
      }
    }
    return isEqual
  }

}

Scala Bean


package myPackage

import myPackage.mySubPackage

case class Person ( val givenName : String , val familyName : String ) {
  def toString() : String = {
    s"${givenName} ${familyName}"
  }
}
